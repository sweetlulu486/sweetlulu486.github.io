## 링크

[BOJ14888 연산자 끼워넣기](https://www.acmicpc.net/problem/14888)



## 문제에 대한 정보

- 수 n
- 연산자의 수 n-1
   
- 덧셈 0, 뺄셈 1, 곱셈 2, 나눗셈 3
  - 0 0 1 0 인 경우 곱셈이 1개 있다.
  - 0 1 1 0 인 경우 뺄셈 1, 곱셈 1개가 있다.
  - 0 1 2 0 인 경우 뺄셈 1, 곱셈 2개가 있다.
  - 2 1 1 1 인 경우 예상 출력 결과는 00123
- 연산자의 우선순위는 없다.

## 소스코드

```java
import java.io.*;
import java.util.*;

public class Main {

	public static boolean next_permutation(int[] arr) {
		int i = arr.length - 1;
		//i가 0일 경우 i-1가 음수가 되므로 i > 0여야 한다.
		//arr[i-1]≥arr[i]를 만족하지 않는 i-1와 i번째를 찾으면 된다.
		//만족하지 않아야 반복문을 진행시키므로
		while(i > 0 && arr[i-1] >= arr[i]) i--;
		if(i == 0) return false;

		int j = arr.length - 1;
		//arr[i-1] ≥ arr[j] 를 만족하지 않는 값을 찾으면 된다.

		//j를 찾는다.
		while(arr[i-1] >= arr[j]) j--;

		//arr[i-1]과 arr[j]를 swap한다.
		int tmp = arr[i-1];
		arr[i-1] = arr[j];
		arr[j] = tmp;

		//array reverse
		int k = arr.length - 1;
		while(i < k) {
			tmp = arr[i];
			arr[i] = arr[k];
			arr[k] = tmp;
			i++; k--;
		}

		return true;

	}


	static int[] arr;
	static int[] opers;

	public static int calc(int[] arr, int[] opers) {

		int ret = arr[0];
		for(int i = 1; i < arr.length; i++){
			switch(opers[i-1]) {

			// 0, 1, 2, 3 = 덧셈, 뺄셈, 곱셈, 나눗셈
			case 0 :
				ret += arr[i];
				break;

			case 1 :
				ret -= arr[i];
				break;

			case 2 :
				ret *= arr[i];
				break;

			case 3 :
				ret /= arr[i];
				break;

			}

		}

		return ret;
	}

	public static void main(String[] args) throws Exception {

		Scanner sc = new Scanner(System.in);

        /*
		예제 입력 1
		2 // 입력받는 수의 갯수
		5 6 // 입력 수열
		0 0 1 0 // 덧셈 뺄셈 곱셈 나눗셈
		*/

        int n = sc.nextInt();

        //수 n
        //연산자의 수 n-1
        arr = new int[n];
        opers = new int[n-1];

        //n개의 수 입력받는 곳
        for(int i = 0; i < n; i++) {
        	arr[i] = sc.nextInt();
        }

        //덧셈 0, 뺄셈 1, 곱셈 2, 나눗셈 3
        //0 0 1 0 인 경우 곱셈이 1개 있다.
        //0 1 1 0 인 경우 뺄셈 1, 곱셈 1개가 있다.
        //0 1 2 0 인 경우 뺄셈 1, 곱셈 2개가 있다.
        //2 1 1 1 인 경우 예상 출력 결과는 00123
        //연산자의 우선순위는 없다.

        //뺄셈을 나눗셈으로 하는 경우 양수로 바꾼 뒤 몫을 취하고, 그 몫을 음수로 바꾼 것과 같다.
        int cnt = 0;
        for(int i = 0; i < 4; i++) {
        	int operCnt = sc.nextInt();

        	//0 1 2 0 인 경우 뺄셈 1, 곱셈 2개가 있다.
        	//뺄셈의 경우 1번 실행 되고 i(1)가 ArrayList에 1번 들어간다.
        	//곱셈의 경우 2번 실행 되고 i(2)가 ArrayList에 2번 들어간다.
        	for(int p = 0; p < operCnt; p++){
        		opers[cnt] = i;
        		cnt++;
        	}
        }

        ArrayList<Integer> result = new ArrayList<>();
        do{
        	int tmp = calc(arr, opers);
        	result.add(tmp);
        } while(next_permutation(opers));

        Collections.sort(result);
        System.out.println(result.get(result.size() - 1));
        System.out.println(result.get(0));

        sc.close();
	}

}
```
